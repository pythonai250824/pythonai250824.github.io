[2a] Import Functions Demo

global_func.py

# Define global functions
def hello():
    print('hello')

def get_int(msg: str = 'Enter number: ', err: str = 'Something went wrong.... try again'):
    """
    This function gets an input of integer from the user
    whenever a string is received it will get it again

    Parameters:
        msg: str, default 'Enter number: '
        err: str = 'Something went wrong.... try again'

    Returns:
        integer number
    """
    num: int = None

    while not num:
        try:
            num = int(input(msg))
        except:
            print(err)
    return num

if __name__ == "__main__":
    print('testing import ...')

This program defines two global functions: `hello`, which prints a greeting, and `get_int`, which prompts the user for an integer input and handles invalid inputs gracefully.

[3] Prime Number Check

Python Program

# Check if a number is prime
prime: bool = True
num: int = int(input('Enter number: '))
for i in range(2, num):
    if num % i == 0:
        prime = False
        break
print(f"{num} is {'Not ' if not prime else ''}Prime")

# Check divisibility of number
divided: [bool] = []
for i in range(2, num):
    is_divided: bool = num % i == 0
    divided.append(is_divided)
print(divided)
comp = [num % i == 0 for i in range(2, num)]
print(comp)
print(f"{num} is {'Not ' if any(divided) else ''}Prime")

print(f"{num} is {'Not ' if any(num % i == 0 for i in range(2, num)) else ''}Prime")
print([num % i for i in range(2, num)])
print(f"is Prime? {all([num % i for i in range(2, num)])}")

[4] Sum of Digits

Python Program

# Calculate the sum of digits of a number
num: int = int(input('Enter number: '))
sum_digits: int = 0
while num > 0:
    rdigit: int = num % 10
    sum_digits += rdigit
    num = num // 10
print(f'digit sum = {sum_digits}')
