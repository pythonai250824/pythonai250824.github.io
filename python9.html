What is Lambda

Python Program

                
# Lambdas are anonymous functions defined with the lambda keyword
# Example of a regular function
def only_positive(x):
    return x < 0  # Return True if x is negative, else False

# The same function using lambda
only_positive_with_lambda = lambda x: x < 0  # create a function using Lambda

print(only_positive_with_lambda(-12))  # Output: True

Filter using lambda

Python Program

                
# Filtering lists using filter and lambdas

# Filter positive numbers
print("filter", list(filter(lambda x: x > 0, [-4, 10, 0, -1, -1, 20])))  # Output: [10, 20]

# Function to check if a number is even
def is_zugi(x):
    return x % 2 == 0  # Return True if x is even, else False

print("filter zugi", list(filter(is_zugi, [-2, 0, 4, 1, -30, 400, 11])))  # Using filter with is_zugi
print("filter zugi lambda", list(filter(lambda x: x % 2 == 0, [-2, 0, 4, 1, -30, 400, 11])))  # Using lambda for the same task

# Filtering non-zero items
print("filter non-zero", list(filter(lambda x: x != 0, [-2, 0, 4, 1, -30, 400, 11])))  # Filter non-zero items

# Bonus: filter strings that start with 'a' or 'A'
print("filter starts with a", list(filter(lambda word: word.upper().startswith("A"), ["apple", "anaconda", "banana"])))

# Bonus: filter strings longer than 3 characters
print("filter more than 3 letters", list(filter(lambda word: len(word) > 3, ["oz", "danny", "moshe", "roy", "sharona"])))

# Lazy evaluation with filter
list_1 = [-1, -2, 0, -10]
filter_me = filter(lambda x: x > 0, list_1)  # Filter positive numbers
print(filter_me)  # Filter object
list_1.append(100)  # Modify original list
print(list(filter_me))  # Output: [100]

Map using lambda

Python Program

                
# Map examples to transform lists

# Square each number in the list
print('map x ** 2', list(map(lambda x: x ** 2, [-2, 0, 4, 1, -30, 400, 11])))  # Output: [4, 0, 16, 1, 900, 160000, 121]

# Convert strings to integers
print('map int', list(map(int, ["-2", "0", "4", "1", "-30", "400", "11"])))  # Output: [-2, 0, 4, 1, -30, 400, 11]

# Map to return each number // 10
print('map x // 10', list(map(lambda x: x // 10, [-2, 0, 4, 1, -30, 400, 11])))  # Output: [0, 0, 0, 0, -3, 40, 1]

# Map to label even numbers as 1 and odd as 0
print('map x % 2', list(map(lambda x: 1 if x % 2 == 0 else 0, [-2, 0, 4, 1, -30, 400, 11])))  # Output: [1, 1, 1, 0, 0, 1, 0]

# Map to label even numbers as "zugi" and odd as "e-zugi"
print('map zugi ezugi', list(map(lambda x: "zugi" if x % 2 == 0 else "e-zugi", \
                   [-2, 0, 4, 1, -30, 400, 11])))  
# ------> Output: ['zugi', 'zugi', 'zugi', 'e-zugi', 'e-zugi', 'zugi', 'e-zugi']

Sort using key function

Python Program

                
# Sorting examples using sorted and lambda functions

words = ["zzzzz", "t", "abc", "cccc", "bb", "b"]
print(words)  
print(sorted(words))  # ['abc', 'b', 'bb', 'cccc', 't', 'zzzzz']

print(sorted(words, key=lambda word: len(word)))  # Sort by length of word
print(sorted(words, key=lambda word: (len(word), word)))  # Sort by length, then alphabetically

def key_func(word):
    return len(word)
print(sorted(words, key=key_func))  # Using key_func to sort by length

# Sorting a list of lists
data = [["danny", 12], ["bob", 30], ["aaron", 42]]
print("sort by age", sorted(data, key=lambda list_item: list_item[1]))  # Sort by age
print("sort by name", sorted(data, key=lambda list_item: list_item[0]))  # Sort by name

# Sorting with negative keys and multiple keys
print(sorted([1, 5, 10, -100, 20, 0], key=lambda x: -1 * x))  # Sort in descending order
print("one key", sorted([1, 5, 10, -100, 20, 3, 1, 0], key=lambda x: 1 if x % 2 == 0 else 0))  # Sort by even/odd
print("two key", sorted([1, 5, 10, -100, 20, 3, 1, 0], key=lambda x: (1 if x % 2 == 0 else 0, x)))  # Sort by even/odd, then value

Scopes

Python Program

                
x = 10

def new_x():
    x = 999 # when not using global this creates a new x for the function scope
    print(f"new_x {x}")  

def arg_x(x: int):
    x = 888 # when argument in the same name it creates a new x for the function scope
    print(f"arg_x {x}")  

def print_x(x):
    # bad practice to use global variables
    # variables should be declared inside the function or passed as parameters
    print(f"print_x {x}")  # when not using global-- x here is read-only from the global scope

def change_x():
    # bad practice to use global variables
    # variables should be declared inside the function or passed as parameters
    global x
    x = x + 1 # after global declared, we can change the x of the global scope
    print(f"change_x {x}")  

new_x()
arg_x(x)
print_x(2)
change_x()
print('last print', x)

LeetCode Two Sum

Python Program

                
# Question:
# Given an array of integers nums and an integer target, return indices of the two numbers such that they add up to target.
# You may assume that each input would have exactly one solution, and you may not use the same element twice.
# Example 1:
# Input: nums = [2,7,11,15], target = 9
# Output: [0,1]
# Explanation: Because nums[0] + nums[1] == 9, we return [0, 1].
# Example 2:
# Input: nums = [3,2,4], target = 6
# Output: [1,2]
# Example 3:
# Input: nums = [3,3], target = 6
# Output: [0,1]
# You can return the answer in any order.

def twoSum(nums, target):
    for i in range(len(nums) - 1):
        for j in range(i + 1, len(nums)):
            if nums[i] + nums[j] == target:
                return [i, j]
    return None

print(twoSum([2, 7, 11, 15], 9))  # [0, 1]
print(twoSum([3, 2, 4], 6))  # [1, 2]
