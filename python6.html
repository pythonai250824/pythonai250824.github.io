# Working with lists

numbers: list[int] = [10, 2, -1, 4]
print(numbers)
print(f"numbers[0] = {numbers[0]}")  # 10
print(f"numbers[3] = {numbers[3]}")  # 4

# Iterating with index
for i in range(0, len(numbers)):
    print(f"numbers[{i}] = {numbers[i]}", end=" | ")
print()

# Iterating with for-each
for number in numbers:
    print(number, end=" ")
print()

# Slicing
print(f"numbers[0: 3] = {numbers[0: 3]}")  # [10, 2, -1]
print(f"numbers[:3] = {numbers[:3]}")  # [10, 2, -1]
print(f"numbers[:] = {numbers[:]}")  # [10, 2, -1, 4]
# numbers[3:] == from index 3 till the end of the list
# numbers[::-1] == reverse
# numbers[::2] == skip by 2

# Updating elements
numbers[0] = 200
print('after numbers[0] = 200', numbers)
numbers.insert(2, 444)
print('after numbers.insert(2, 444)', numbers)

# Input grades from user until -999 is entered
grades: list[int] = []
while True:
    grade: int = int(input("enter grade [-999 to exit]:"))
    if grade == -999:
        break
    if grade < 0 or grade > 100:
        continue
    grades.append(grade)

print(grades)
print(f"max(grades) = {max(grades)}")
print(f"min(grades) = {min(grades)}")
print(f"sum(grades) = {sum(grades)}")
print(f"len(grades) = {len(grades)}")
print(f"avg(grades) = {sum(grades) / len(grades) : .2f}")

import statistics
print(f"statistics.mean(grades) = {statistics.mean(grades) : .2f}")

Python Program


# More list operations

numbers: list[int] = [10, 2, -1, 4]
more_numbers: list[int] = [99, -999]

numbers.extend(more_numbers)  # Add list 2 to list 1
print('after concat', numbers)

# Remove first occurrence of an item
numbers.remove(-999)
print('after remove -999', numbers)

# Pop last item
pop: int = numbers.pop()
print('pop last', pop)
print('after pop', numbers)

# Pop item at index 1
pop2: int = numbers.pop(1)
print('pop index [1] ==', pop2)
print('after pop [1]', numbers)

# Remove item at index 1
del numbers[1]
print('after remove item index 1', numbers)

# Remove all occurrence of 99
while 99 in numbers:
    numbers.remove(99)

# Clear all items
numbers.clear()
print('after clear', numbers)

numbers.extend([10, 2, -1, 4])
print('after extend', numbers)
print('numbers.index(-1)', numbers.index(-1))

numbers.extend([4, 4, 4, 4, -9, 12, 4])
print('after extend', numbers)
print('numbers.count(4)', numbers.count(4))

# Sort and reverse
clone = numbers.copy()
clone.sort()
print('after sort [clone]', clone)
clone.sort(reverse=True)
print('after sort reverse [clone]', clone)

# Sorted does not change the list
print('after sorted', sorted(numbers))
print('after sorted reverse', sorted(numbers, reverse=True))
print('after list + reversed', list(reversed(numbers)))  # iterator

print(numbers)

# Boolean operations on lists
bool1: list[bool] = [True, True, True]
bool2: list[bool] = [True, True, False]
bool3: list[int] = [20, 0, 300, 100, 100]
print('all bool1', bool1, all(bool1))  # True
print('all bool2', bool2, all(bool2))  # False
print('all bool3', bool3, all(bool3))  # False

bool1: list[bool] = [True, True, True]
bool2: list[bool] = [True, True, False]
bool3: list[int] = [20, 0, 300, 100, 100]
bool4: list[bool] = [False, False]
bool5: list[int] = [0, 0, 0, 0, 0]
print('any bool1', bool1, any(bool1))  # True
print('any bool2', bool2, any(bool2))  # True
print('any bool3', bool3, any(bool3))  # True
print('any bool4', bool4, any(bool4))  # False
print('any bool5', bool5, any(bool5))  # False

# List comprehensions

n_list: list[int] = []
for i in range(0, 10):
    n_list.append(i ** 2)
print(n_list)

# Using list comprehension
n_list = [i ** 2 for i in range(0, 20)]
print(n_list)

# Comprehension to create a list of even numbers
targil1: list[int] = [i for i in range(0, 21, 2)]
print(targil1)

# Comprehension to create a list of 10 random numbers
import random
targil2: list[int] = [random.randint(1, 100) for _ in range(10)]
print(targil2)

# Comprehension to create a list of 10 random even numbers
just_random: list[int] = [random.randint(1, 100) for _ in range(10)]
random_even: list[int] = [n for n in just_random if n % 2 == 0]
print(random_even)

# Error handling in Python

numbers: list[int] = [1, 4, 5, 9]

# Potential error
try:
    grade: int = int(input('Enter grade:'))
    print('grade is ok')
except:
    print('something went wrong ...')

# Loop until valid input
grade: int = None
while not grade:
    try:
        grade = int(input('Enter grade:'))
        print('grade is ok')
    except:
        print('something went wrong ... try again')

# Print the error
grade: int = None
while not grade:
    try:
        grade = int(input('Enter grade:'))
        print('grade is ok')
    except Exception as e:
        print(f'something went wrong --{e}-- ... try again')

print('finish')

# Drawing stars in a pyramid shape

number: int = 15
space: int = number // 2
for i in range(1, number + 1, 2):
    print(space * " ", end="")
    for j in range(1, i + 1):
        print("*", end="")
    print()
    space -= 1
