modulo and div

Python Program

                
# Get a number from user
number: int = int(input("Give me a number?"));

# Perform division and modulo operations
print(f"{number} / 2 = {number / 2}")
print(f"{number} % 2 = {number % 2}")

# Check if the number is even or odd using modulo operator
if number % 2 == 0: # is the remainder when divided by 2 equal to 0?
    print(f"{number} is Zugi even");
else:
    print(f"{number} is E-Zugi odd");

While Loop

Python Program

x: int = 1
while x < 20:  # condition is tested before entering 1st cycle
    print(x, end=" ")
    x = x + 1  # increment x by 1
print('finish')  # loop finished
