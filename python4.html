Do While

Python Program

# input 3 numbers in a loop a, b, c until a * 2 + b > c - 1
a: int
b: int
c: int = 0

while True:
    a = float(input('enter a: '))
    b = float(input('enter b: '))
    c = float(input('enter c: '))
    print(f"a: {a} b: {b} c: {c}")
    if a * 2 + b <= c - 1:  # break condition
        break

print('finish')  # loop finished

price: int = 0
total: int = 0
number_of_products: int = 0

while True:
    price = float(input('enter product price: '))
    total = total + price
    number_of_products = number_of_products + 1
    if total > 1000:  # break condition
        break

diff: float = total - 1_000
avg: float = total / number_of_products
print(f"you bought {number_of_products} products.")
print(f"the avg is: {avg}.")
print(f"you exceed the 1,000 by {diff}")

For and Foreach Loop

Python Program

for i in range(1, 11):
    print(i, end=" ")

total: int = 0
for i in range(1, 11):  # loop from 1 to 10
    price: int = int(input('enter price: '))
    total = total + price
    if total > 1000:  # break condition
        break

# for-each loop over a string
for c in "Hello":
    print(c, end="")
print()
# loop with step 2
for i in range(1, 20, 2):
    print(i, end=" ")
print()
# loop with negative step
for i in range(20, 1, -1):
    print(i, end=" ")
print()
# simple loop from 1 to 19
for i in range(1, 20, 1):
    print(i, end=" ")
